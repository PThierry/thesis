\newglossaryentry{ip}{name=IP,description={Internet Protocol}}
\newglossaryentry{atm}{name=ATM,description={Protocole de commutation de circuit, recouvrant les couches liaison, réseau et transport.}}

\newglossaryentry{diffserv}{name={DiffServ},description={Champs de l'en-tête IP permettant de définir une différenciation de service en fonction de sa valeur}}
\newglossaryentry{qos}{name={QoS},description={{\it Quality Of Service}, ensemble de moyens informatique permettant de fournir une meilleure garantie pour le transport d'un flux.}}

\newglossaryentry{gpp}{name=GPP,description={{\it General Purpose Unit}, general purpose
processing unit containing the CPU core(s), the cache controller(s) and the associated blocks like
the ICU}}

\newglossaryentry{apic}{name=APIC,description={{\it Advanced Programable Interruption Controller},
hardware controller managing associated interrupt lines, allowing advanced control and per-line
dynamic activation/deactivation}}

\newglossaryentry{nmi}{name=NMI,description={{\it Non-Maskable Interrupt}, Interrupt requests which
can't be deactivated by a software configuration of the ICU}}

\newglossaryentry{wcet}{name=WCET,description={{\it Worst Case Execution Time}, temps maximum
nécessaire à l'exécution d'une tache donnée}}
\newglossaryentry{switch}{name=switch, description={matériel réseau faisant une
décapsulation niveau liaison des paquets qu'il reçoit, afin de les rerouter sur la bonne
interface, grâce à une capacité de mémorisation de la position des n{\oe}uds lui étant connectés}}

\newglossaryentry{rfc}{name={RFC},description={{\it Request For Comments}}}

\newglossaryentry{mmu}{name=MMU,description={{\it Memory Management Unit}, Unité matérielle intégrée au processeur, permettant de fournir une capacité
d'adressage logique de la mémoire, ainsi qu'un certain nombre de protections en terme de gestion d'accès.}}
\newglossaryentry{iommu}{name=I/OMMU,description={{\it In/Out Memory Management Unit}, Unité matérielle intégrée au processeur, permettant de contrôler
les accès des éléments matériel (hors CPU) à la mémoire}}

\newglossaryentry{taskset}{name={Taskset},description={Ensemble de taches. Définit une liste exhaustive de taches temps réel, avec un certain nombre de propriété comme la période d'instanciation, le WCET ou la {\it deadline}}}
\newglossaryentry{wcep}{name={WCEP}, description={Pour une tache donnée, définit le chemin d'exécution le plus long}}
